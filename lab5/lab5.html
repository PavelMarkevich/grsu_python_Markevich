import cv2
import numpy as np

img = cv2.imread('image.jpg',1)
(h,w,c) = img.shape
parts = []
step_x = 2
step_y = 1
for x in range(step_x):
    for y in range(step_y):
        xratio1 = x/step_x
        xratio2 = (x+1)/step_x
        yratio1 = y/step_y
        yratio2 = (y+1)/step_y
        part = img[int(yratio1*h):int(yratio2*h),int(xratio1*w):int(xratio2*w)].copy()
        parts.append(part)
cv2.imwrite("flip1.png", parts[0])
cv2.imwrite("flip2.png", parts[1])
cv2.waitKey(0)
